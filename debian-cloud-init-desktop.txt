#cloud-config

# 1) Update and upgrade the system
package_update: true
package_upgrade: true

# Install base packages
packages:
  - xfce4
  - xfce4-goodies
  - xrdp
  - firefox-esr
  - wget
  - curl
  - gpg
  - apt-transport-https

runcmd:
  # Ensure the keyrings directory exists
  - mkdir -p /etc/apt/keyrings

  # 5) Setup VS Code Repository
  - sh -c 'curl -sSL https://packages.microsoft.com/keys/microsoft.asc | gpg --dearmor --yes -o /etc/apt/keyrings/packages.microsoft.gpg'
  - sh -c 'echo "deb [arch=amd64,arm64,armhf signed-by=/etc/apt/keyrings/packages.microsoft.gpg] https://packages.microsoft.com/repos/code stable main" > /etc/apt/sources.list.d/vscode.list'

  # 6) Setup GitHub Desktop Repository
  - sh -c 'curl -sSL https://apt.packages.shiftkey.dev/gpg.key | gpg --dearmor --yes -o /usr/share/keyrings/shiftkey-packages.gpg'
  - sh -c 'echo "deb [arch=amd64 signed-by=/usr/share/keyrings/shiftkey-packages.gpg] https://apt.packages.shiftkey.dev/ubuntu/ any main" > /etc/apt/sources.list.d/shiftkey-packages.list'

  # Update apt cache for newly added repositories and install the tools
  - apt-get update -y
  - DEBIAN_FRONTEND=noninteractive apt-get install -y code github-desktop

  # 7) Install Codex CLI using Special Script
  - sh -c 'curl -fsSL https://raw.githubusercontent.com/svk27/Project-Mango/refs/heads/main/Layer2Setup/codex_setup.sh | bash'
